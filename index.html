        function filterByBreed(breed) {

            currentBreed = breed;

            const buttons = document.querySelectorAll('#breed-filters .category-btn');

            buttons.forEach(btn => {

                btn.classList.remove('active');

                if(btn.innerText.includes(breed === 'all' ? '전체' : (breed === 'small' ? '소형견' : (breed === 'medium' ? '중형견' : '대형견')))) {

                    btn.classList.add('active');

                }

            });

            renderShops();

        }


        function filterByStyle(style) {

            currentStyle = (currentStyle === style) ? '' : style; // 토글

            const buttons = document.querySelectorAll('.style-btn');

            buttons.forEach(btn => {

                if(btn.innerText.includes(style) && currentStyle !== '') {

                    btn.classList.add('bg-indigo-600', 'text-white');

                    btn.querySelector('i').classList.replace('text-indigo-500', 'text-white');

                } else {

                    btn.classList.remove('bg-indigo-600', 'text-white');

                    btn.querySelector('i').classList.replace('text-white', 'text-indigo-500');

                }

            });

            renderShops();

        }
